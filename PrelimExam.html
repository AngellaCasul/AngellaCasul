Prelim Grade Calculator

from pyscript import Element def calculate_grades(event): try: # Get the number of absences of the student absences = int(Element("attendance").element.value) if absences >= 4: Element("result").write("FAILED due to excessive absences.") Element("result").element.style.display = "block" return # Calculate the attendance score (starting from 100, subtracting 10 for each absence) attendance_score = 100 - (absences * 10) # Calculate the other grades based on input recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 prelim_grade = round((attendance_score * 0.1) + overall_performance + prelim_exam, 2) # First required grade calculation (75% pass) prelim = prelim_grade # Use the prelim grade calculated above passing_grade = 75 deans_lister_grade = 90 prelim_percent = 0.20 midterm_percent = 0.30 final_percent = 0.50 current_total = prelim * prelim_percent required_total_75 = passing_grade - current_total required_total_90 = deans_lister_grade - current_total # Calculate required midterm and final grades for 75% and 90% required_midterm_and_final_75 = round(required_total_75 / (midterm_percent + final_percent), 2) required_midterm_and_final_90 = round(required_total_90 / (midterm_percent + final_percent), 2) # Prepare result text for prelim grade result_text = f"Your Prelim Grade is: {prelim_grade:.2f}" if prelim_grade >= 90: result_text = f"Your Prelim Grade is: {prelim_grade:.2f}. Congratulations! You are qualified for Dean's Lister." else: result_text = f"Your Prelim Grade is: {prelim_grade:.2f}. You are not qualified for Dean's Lister." # Add required grades for passing and Dean's Lister result_text += f" You need an average of {required_midterm_and_final_75:.2f} in midterms and finals to pass (75%)." result_text += f" You need an average of {required_midterm_and_final_90:.2f} in midterms and finals to qualify for Dean's Lister (90%)." Element("result").write(result_text) Element("result").element.style.display = "block" except ValueError: Element("result").write("Invalid input. Please enter numbers only.") Element("result").element.style.display = "block" Element("calculateButton").element.onclick = calculate_grades
