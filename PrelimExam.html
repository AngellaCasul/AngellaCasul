Prelim Grade Calculator

from pyscript import Element def calculate_grades(event): try: # Get the number of absences absences = int(Element("attendance").element.value) if absences >= 4: Element("result").write("FAILED due to excessive absences.") Element("result").element.style.display = "block" return # Calculate attendance score (starting from 100, subtracting 10 for each absence) attendance_score = 100 - (absences * 10) # Calculate other grades based on input recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 prelim_grade = round((attendance_score * 0.1) + overall_performance + prelim_exam, 2) result_text = f"Your Prelim Grade is: {prelim_grade:.2f}" Element("result").write(result_text) Element("result").element.style.display = "block" except ValueError: Element("result").write("Invalid input. Please enter numbers only.") Element("result").element.style.display = "block" Element("calculateButton").element.onclick = calculate_grades
